La Coercition

Le moteur JavaScript change automatiquement le type des données pour essayer de faire l'opération :


5 + null    // retourne 5         null est convertit en 0
"5" + null  // retourne "5null"   null est convertit en "null"
"5" + 2     // retourne 52        2 est convertit en "2"
"5" - 2     // retourne 3         "5" est convertit en 5
"5" * "2"   // retourne 10        "5" et "2" sont convertit en 5 et 2

Les objets

Une entité qui possède des properiétés (variables) qui lui sont attachées.

// syntaxe litérale
var voiture = {couleur: "route"};    

// création basée sur un prototype
var voiture2 = Object.create(voiture); 

// définition d'un constructeur
function Moto(couleur) {this.couleur = couleur;} 
// création d'une nouvelle instance.
var motoRouge = new Moto('rouge');

Les prototypes

Le prototype est un objet partagé par toutes les instances du constructeur

function Moto(couleur) {this.couleur = couleur;} 

var motoRouge = new Moto('rouge'); 

console.log(motoRouge.type); // undefined

Moto.prototype.type = 'sportive';

var motoBleu = new Moto('bleu');
motoBleu.type = 'roadster';

console.log(motoRouge.type); // "sportive"
console.log(motoBleu.type); // "roadster"

Les Fonctions : First-Class Citizen !

Une fonction est un object, instance de Function

Hérite donc de Function.prototype

  • apply(), call(), bind(), toString() et length

Comme les autres objets elle peut avoir ses propres propriétés

Une fonction c'est :

  • le mot clé 'function'
  • un nom (optionnel => fonction annonyme)
  • des paramètres (optionnel et illimité)
  • un corps entre accolades

Les fonctions fléchés

var multiplication = function(x) {
	return x * x;
}
multiplication(5); // 25

var multiplication2 = x => x * x;
multiplication2(5); // 25

var add = (x,y) => x + y;
add(2,3); //5

pas de redéfinition de 'this'

Les variables et le scope

Toujours utiliser le mode strcit: `'use strict';`.

Une variable, déclarée avec le mot-clé `var`, à comme portée le contexte le courant.

'use strict';
function fa() { var a = 1; }
a = 2; // ReferenceError: assignment to undeclared variable a

var b = 5;
function fb() { 
	var c = 3;
	console.log(b, c); // 5 3
	function fc() { 
		b = 2; 
		c = 6;
	}
	fc();
	console.log(b, c); // 2 6
}
fb();
console.log(typeof c); // undefined

Les closures

function creerAdditionneur(a) {
  return function(b) {
    return a + b;
  }
}
var additionner5 = creerAdditionneur(5);
var additionner20 = creerAdditionneur(20);
additionner5(6); // ?
additionner20(7); // ?

Une fermeture permet d'associer des données (l'environnement) avec une fonction qui agit sur ces données.

Peut également servir à encapsuler des variables pour les rendre privées.

Notation utile car elle permet de "modulariser" son code.

 // moteur.js
(function(exports) {
  "use strict";
  function Moteur(carburant) { this.carburant = carburant || 'diesel'; }
  exports.Moteur = Moteur;
})(this);
 // moto.js
(function(exports) {
  "use strict";
  function Moto(couleur) {this.couleur = couleur;} 
  exports.Moto = Moto;
  Moto.prototype = new Moteur();
})(this);
